Firestore Rules Addition for QR Code Access Tokens

Add these rules to your existing Firestore security rules:

// Allow access to access_tokens collection for QR code functionality
match /access_tokens/{tokenId} {
  allow read, write: if request.auth != null;
}

Complete rules should look like this:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      allow read, write: if request.auth != null && exists(/databases/$(database)/documents/users/$(request.auth.uid)) && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == "Super Admin";
    }
    
    // Allow TMO to read users in their township
    allow read: if request.auth != null && exists(/databases/$(database)/documents/users/$(request.auth.uid)) && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == "TMO" && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.township == resource.data.township;
    
    // Allow users to read patients (for their role/township)
    match /patients/{patientId} {
      allow read, write: if request.auth != null;
    }
    
    // Allow users to read/write records subcollection
    match /patients/{patientId}/records/{document=**} {
      allow read, write: if request.auth != null;
    }
    
    // Allow users to read/write medication subcollection
    match /patients/{patientId}/medication/{document=**} {
      allow read, write: if request.auth != null;
    }
    
    // Allow access to testRecords subcollection
    match /patients/{patientId}/testRecords/{testId} {
      allow read, write: if request.auth != null;
    }
    
    // Allow access to postpartum_visits subcollection
    match /patients/{patientId}/postpartum_visits/{visitId} {
      allow read, write: if request.auth != null;
    }
    
    // Allow users to read/write antenatal_visits subcollection
    match /patients/{patientId}/antenatal_visits/{document=**} {
      allow read, write: if request.auth != null;
    }
    
    // Allow users to read/write plotData subcollection
    match /patients/{patientId}/plotData/{document=**} {
      allow read, write: if request.auth != null;
    }
    
    // NEW: Allow access to access_tokens collection for QR code functionality
    match /access_tokens/{tokenId} {
      allow read, write: if request.auth != null;
    }
  }
}
